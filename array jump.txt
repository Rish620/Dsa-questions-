class Solution {
    public boolean canJump(int[] nums) {
        int n=nums.length;
        int maxlevel=0;
        for(int i=0; i<n; i++)
        {
            if(i>maxlevel)
            return false;
            maxlevel=Math.max(maxlevel,i+nums[i]);
        }
        return true;
    }
}
